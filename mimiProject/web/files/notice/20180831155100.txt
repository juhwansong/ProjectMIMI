PROMPT CREATING TB_BOARD_NOTICE
-- 공지사항게시판
------------------------------------------------------------------------------------
CREATE TABLE TB_BOARD_NOTICE (
    NOTICE_NO          VARCHAR2(20) CONSTRAINT PK_NOTICE_NO PRIMARY KEY, -- 공지사항번호 PK
    USER_ID            VARCHAR2(20)  NOT NULL, -- 작성자
    NOTICE_TITLE       VARCHAR2(100)  NOT NULL, -- 공지사항제목
    NOTICE_CONTENTS    VARCHAR2(300) NOT NULL, -- 공지사항내용

);

COMMENT ON COLUMN TB_BOARD_NOTICE.USER_ID IS '작성자';
COMMENT ON COLUMN TB_BOARD_NOTICE.NOTICE_TITLE IS '공지사항제목';
COMMENT ON COLUMN TB_BOARD_NOTICE.NOTICE_CONTENTS IS '공지사항내용';
COMMENT ON COLUMN TB_BOARD_NOTICE.NOTICE_DATE IS '공지사항작성일';
COMMENT ON COLUMN TB_BOARD_NOTICE.NOTICE_REAL_FILE IS '원본파일명';
COMMENT ON COLUMN TB_BOARD_NOTICE.NOTICE_RENAME_FILE IS '바뀐파일명';

CREATE TABLE TB_USER (
   USER_ID     VARCHAR2(20) CONSTRAINT PK_USER_ID PRIMARY KEY, -- 아이디 PK
   USER_PWD    VARCHAR2(200)NOT NULL, -- 비밀번호
   USER_NAME   VARCHAR2(20) NOT NULL, -- 이름
   NICKNAME    VARCHAR2(30) NOT NULL, -- 닉네임

   BIRTH       VARCHAR2(20) NULL,     -- 생년월일

CREATE TABLE TB_BOARD_REVIEW (
   BOARD_NO       VARCHAR2(20) CONSTRAINT PK_BOARD_NO PRIMARY KEY, -- 글번호 PK
   CATEGORY_NO    VARCHAR2(20)  NOT NULL  CONSTRAINT FK_BR_CATEGORY_NO REFERENCES TB_CATEGORY ON DELETE CASCADE, -- 카테고리번호FK
   USER_ID        VARCHAR2(20)  NOT NULL CONSTRAINT FK_BR_USER_ID REFERENCES TB_USER ON DELETE CASCADE, -- 작성자FK

   TITLE          VARCHAR2(100)  NOT NULL, -- 제목
   CONTENTS       CLOB          NOT NULL, -- 내용 텍스트
  CONTENTS_TAG   CLOB          NOT NULL, -- 내용 태그 포함
   BOARD_DATE     DATE           DEFAULT SYSDATE NOT NULL, -- 작성일
  COMMENT_NUM    NUMBER        DEFAULT 0 NOT NULL, -- 댓글 수
   SHOP_NAME      VARCHAR2(50)  NOT NULL, -- 매장이름
   SHOP_ADDRESS   VARCHAR2(200)  NOT NULL, -- 매장주소
   SHOP_CALL      VARCHAR2(20)  NULL,     -- 매장연락처

   LONGITUDE      NUMBER        NOT NULL, -- 경도

   RECOMMEND      NUMBER        DEFAULT 0 NOT NULL, -- 추천수
   FILE_CNT       NUMBER        NULL,     -- 파일개수
   BOARD_LINK     VARCHAR2(20)  NOT NULL, -- 게시물주소
   STATE          VARCHAR2(10) DEFAULT 'SN' CONSTRAINT CHEK_ST_BR CHECK (STATE IN ('SD', 'SN')), -- 상태
   THUMBNAIL_NAME VARCHAR2(200)  NULL,     -- 썸네일명
   DEL_DATE       DATE          NULL      -- 삭제일
  
);

PROMPT CREATING V_NOTICE_LIST 
--공지사항 목록
------------------------------------------------------------------------------------
CREATE OR REPLACE VIEW V_NOTICE_LIST
AS
SELECT NOTICE_NO, NOTICE_TITLE, NICKNAME, NOTICE_DATE 
FROM TB_BOARD_NOTICE NT, TB_USER U
WHERE NT.USER_ID = U.USER_ID;


------------------------------------------------------------------------------------
PROMPT CREATING V_NOTICE_VIEW 
--공지사항 보기
------------------------------------------------------------------------------------
CREATE OR REPLACE VIEW V_NOTICE_VIEW
AS
SELECT NOTICE_NO, NOTICE_TITLE, NICKNAME , NOTICE_DATE, NOTICE_CONTENTS,
       NOTICE_REAL_FILE, NOTICE_RENAME_FILE, FILE_CNT
FROM TB_BOARD_NOTICE NT, TB_USER U
WHERE NT.USER_ID = U.USER_ID;


------------------------------------------------------------------------------------
PROMPT CREATING V_ADMIN_REVIEW_LIST 
--관리자리뷰 목록
------------------------------------------------------------------------------------
CREATE OR REPLACE VIEW V_ADMIN_REVIEW_LIST
AS
SELECT BOARD_NO, TITLE, CONTENTS, BOARD_DATE, CATEGORY_FOOD, CT.CATEGORY_NO CATEGORY_NO,
       COMMENT_NUM, RECOMMEND, THUMBNAIL_NAME, HITS, BOARD_LINK
FROM TB_BOARD_REVIEW BR, TB_USER U, TB_CATEGORY CT
WHERE BR.USER_ID = U.USER_ID
AND BR.CATEGORY_NO = CT.CATEGORY_NO
AND BR.STATE = 'SN'
AND BR.BOARD_GB='AR';


------------------------------------------------------------------------------------
PROMPT CREATING V_ADMIN_REVIEW_VIEW 
--관리자리뷰 보기
------------------------------------------------------------------------------------
CREATE OR REPLACE VIEW V_ADMIN_REVIEW_VIEW
AS
SELECT BOARD_NO, THUMBNAIL_NAME, TITLE, USER_ID, NICKNAME, GRADE_NAME,
       BOARD_DATE, HITS, RECOMMEND, CONTENTS_TAG, CATEGORY_NO, CATEGORY_FOOD,
       SHOP_NAME, SHOP_CALL, SHOP_ADDRESS, LATITUDE, LONGITUDE, TB_BOARD_REVIEW.STATE STATE
FROM TB_BOARD_REVIEW
JOIN TB_CATEGORY USING (CATEGORY_NO)
JOIN TB_USER USING (USER_ID)
WHERE BOARD_GB = 'AR';


------------------------------------------------------------------------------------
PROMPT CREATING V_ADMIN_REVIEW_COMMENT 
--관리자리뷰 댓글 
------------------------------------------------------------------------------------
CREATE OR REPLACE VIEW V_ADMIN_REVIEW_COMMENT
AS
SELECT COMMENT_NO, BOARD_NO, USER_ID, NICKNAME, GRADE_NAME, COMMENT_DATE, COMMENT_CONTENTS, COMMENT_STATE

CREATE OR REPLACE TRIGGER RECOMMEND_INSERT_TRIGGER
AFTER INSERT ON TB_RECOMMEND FOR EACH ROW --추천테이블에 데이터가 추가됐을때
DECLARE 
  SELECTED_ID VARCHAR2(20);  --추천이 추가된 BOARD_NO의 작성자 ID를 담을 변수 선언
  SELECTED_GRADE_POINT NUMBER;--해당 작성자의 등급포인트를 담을 변수 선언
  MAX_GRADE_POINT NUMBER; --IF문을 위해 최고 등급의 MAX_POINT의 값을 담을 변수 선언
BEGIN
  SELECT USER_ID INTO SELECTED_ID FROM TB_BOARD_REVIEW WHERE BOARD_NO = :NEW.BOARD_NO; --추천이 추가된 BOARD_NO의 작성자 ID를 담음
  
  UPDATE TB_BOARD_REVIEW SET RECOMMEND = RECOMMEND+1 --추천수 1증가
  WHERE BOARD_NO =:NEW.BOARD_NO;
  
  UPDATE TB_USER SET GRADE_POINT = GRADE_POINT + 10   --추천 받은 게시물의 작성자 등급 포인트 10 증가
  WHERE USER_ID = SELECTED_ID;
  
  SELECT GRADE_POINT INTO SELECTED_GRADE_POINT FROM TB_USER 
  WHERE USER_ID = SELECTED_ID;--작성자의 증가된 등급포인트를 담음 
  
  SELECT MAX(MAX_POINT) INTO MAX_GRADE_POINT FROM TB_GRADE;
  
  IF MAX_GRADE_POINT >= SELECTED_GRADE_POINT THEN --현재 등급포인트가 가장 높은 등급의 최대치보다 같거나 작을때
    UPDATE TB_USER    --작성자의 등급 이름을 해당 등급포인트에 맞는 이름으로 바꾼다
    SET GRADE_NAME = (SELECT GRADE_NAME FROM TB_GRADE WHERE MIN_POINT <= SELECTED_GRADE_POINT and MAX_POINT >= SELECTED_GRADE_POINT)           
    WHERE USER_ID = SELECTED_ID;  
  END IF;
END;
/

CREATE OR REPLACE TRIGGER RECOMMEND_DELETE_TRIGGER
AFTER DELETE ON TB_RECOMMEND FOR EACH ROW
DECLARE
  SELECTED_ID VARCHAR2(20);  --추천이 추가된 BOARD_NO의 작성자 ID를 담을 변수 선언

BEGIN
  SELECT USER_ID INTO SELECTED_ID FROM TB_BOARD_REVIEW WHERE BOARD_NO = :OLD.BOARD_NO; --추천이 감소된 BOARD_NO의 작성자 ID를 담음
  
  UPDATE TB_BOARD_REVIEW SET RECOMMEND = RECOMMEND-1    --추천수 1 감소

  SELECT MAX(MAX_POINT) INTO MAX_GRADE_POINT FROM TB_GRADE;
  
  IF MAX_GRADE_POINT >= SELECTED_GRADE_POINT THEN --현재 등급포인트가 가장 높은 등급의 최대치보다 같거나 작을때
    UPDATE TB_USER      --작성자의 등급 이름을 해당 등급포인트에 맞는 이름으로 바꾼다

    WHERE USER_ID = SELECTED_ID;

CREATE OR REPLACE PROCEDURE BEST_BOARD_UPDATE_PROC (SELETED_MONTH IN VARCHAR2) --프로시저 생성
IS
  --변수선언부분
  BOARD_NO_ARRAY ARRAY_VARCHAR;
BEGIN 
  --SQL문장, PL/SQL제어 문장   
  SELECT BOARD_NO 
  BULK COLLECT --BULK 처리(루프 없이 BOARD_NO_ARRAY에 SELECT된 값들을 저장한다.)
  INTO BOARD_NO_ARRAY
  FROM
  ( 
  SELECT BOARD_NO
  FROM TB_RECOMMEND
  JOIN TB_BOARD_REVIEW USING(BOARD_NO)
  WHERE TO_CHAR(RECOMMEND_DATE, 'RR/MM') = SELETED_MONTH AND BOARD_GB = 'UR'
  GROUP BY BOARD_NO
  ORDER BY COUNT(BOARD_NO) DESC
  )
  WHERE ROWNUM <= 10;
 
  
  
  FOR I IN 1 .. BOARD_NO_ARRAY.COUNT LOOP
    DBMS_OUTPUT.PUT_LINE(BOARD_NO_ARRAY(I));
    INSERT INTO TB_BOARD_BEST VALUES(BOARD_NO_ARRAY(I), SELETED_MONTH, 'SN', I);
  END LOOP;
  
  COMMIT;
  
END;

CREATE OR REPLACE PROCEDURE DELETE_SD_USER_PROC--프로시저 생성

  USER_ID_ARRAY ARRAY_VARCHAR;
BEGIN

    DELETE FROM TB_RECOMMEND WHERE USER_ID = USER_ID_ARRAY(I);
  END LOOP;
  COMMIT;
  
  FOR I IN 1 .. USER_ID_ARRAY.COUNT LOOP
    DELETE FROM TB_RECOMMEND WHERE BOARD_NO IN (SELECT BOARD_NO FROM TB_BOARD_REVIEW WHERE USER_ID = USER_ID_ARRAY(I));
  END LOOP;
  COMMIT;
  
  FOR I IN 1 .. USER_ID_ARRAY.COUNT LOOP
    DELETE FROM TB_COMMENT_REVIEW WHERE USER_ID = USER_ID_ARRAY(I);
  END LOOP;
  COMMIT;
  
  FOR I IN 1 .. USER_ID_ARRAY.COUNT LOOP
    DELETE FROM TB_COMMENT_REVIEW WHERE BOARD_NO IN (SELECT BOARD_NO FROM TB_BOARD_REVIEW WHERE USER_ID = USER_ID_ARRAY(I));
  END LOOP;
  COMMIT;
  
  FOR I IN 1 .. USER_ID_ARRAY.COUNT LOOP
     DELETE TB_USER WHERE USER_ID = USER_ID_ARRAY(I);
  END LOOP;
  COMMIT;
END;

--라디언스 함수 생성
CREATE OR REPLACE FUNCTION RADIANS(nDegrees IN NUMBER) 
RETURN NUMBER DETERMINISTIC 
IS
BEGIN 
  RETURN nDegrees / 57.29577951308232087679815481410517033235;
END RADIANS;
/


--기준이 되는 위도,경도와 비교할 위도,경도 값을 넣어 지정된 distance 거리안만 찾는 함수
--H_LAT = 기준이 되는 위도, H_LNG = 기준이 되는 경도, T_LAT = 비교할 위도, T_LNG = 비교할 경도
create or replace function DISTNACE_WGS84( H_LAT in number, H_LNG in number, T_LAT in number, T_LNG in number)
return number deterministic
is
begin
  return ( 6371.0 * acos(  
          cos( radians( H_LAT ) )*cos( radians( T_LAT ) )
          *cos( radians( T_LNG )-radians( H_LNG ) )
          +
          sin( radians( H_LAT ) )*sin( radians( T_LAT ) )        
         ));
end DISTNACE_WGS84;
/
